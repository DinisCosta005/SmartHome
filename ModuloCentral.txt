#include <WiFi.h>
#include <WiFiManager.h>
#include <ESPmDNS.h>
#include <PubSubClient.h>

#define RELAY_PIN 5
#define LED_PIN 23
#define RESET_BUTTON_PIN 4
#define FAILSAFE_TIMEOUT 300000

WiFiClient espClient;
PubSubClient client(espClient);
WiFiManager wm;

const char* mqtt_user = "esp_mqtt";
const char* mqtt_pass = "projeto123";
const char* topicState = "casa/central/rele/state";
const char* topicCmd = "casa/central/rele/set";

unsigned long lastConnectedTime = 0;
unsigned long lastReconnectAttempt = 0;
unsigned long tempoBotaoPressionado = 0;
unsigned long ultimoPisca = 0;
bool estadoLed = false;
const int tempoParaReset = 5000;

// ---------------------------------------------------------
// CALLBACK: MODO DE CONFIGURAÇÃO (WIFIMANAGER)
// ---------------------------------------------------------
void configModeCallback(WiFiManager* myWiFiManager) {
  Serial.println("ENTROU NO MODO DE CONFIGURAÇÃO!");
}

// ---------------------------------------------------------
// CALLBACK: RECEBE OS COMANDOS ON/OFF DO HOME ASSISTANT
// ---------------------------------------------------------
void callback(char* topic, byte* payload, unsigned int length) {
  lastConnectedTime = millis();
  String message;
  for (int i = 0; i < length; i++) {
    message += (char)payload[i];
  }

  if (String(topic) == topicCmd) {
    if (message == "ON") {
      digitalWrite(RELAY_PIN, LOW);  // Assumindo relé Activo-LOW (LOW = Liga)
      client.publish(topicState, "ON");
      Serial.println("Comando recebido: Relé LIGADO");
    } else if (message == "OFF") {
      digitalWrite(RELAY_PIN, HIGH);  // HIGH = Desliga
      client.publish(topicState, "OFF");
      Serial.println("Comando recebido: Relé DESLIGADO");
    }
  }
}

// ---------------------------------------------------------
// RECONEXÃO MQTT NÃO-BLOQUEANTE
// ---------------------------------------------------------
boolean reconnect() {
  if (!client.connected()) {
    Serial.println("A procurar o Home Assistant...");
    IPAddress serverIP = MDNS.queryHost("homeassistant");
    client.setServer((serverIP.toString() != "0.0.0.0") ? serverIP : IPAddress(192, 168, 1, 76), 1883);

    if (client.connect("ESP32_Central_Clima", mqtt_user, mqtt_pass)) {
      Serial.println("Ligado ao MQTT (Central)!");
      client.subscribe(topicCmd);
      String currentState = (digitalRead(RELAY_PIN) == LOW) ? "ON" : "OFF";
      client.publish(topicState, currentState.c_str());
      return true;
    }
    Serial.println("Falhou.");
  }
  return client.connected();
}

// ---------------------------------------------------------
// SETUP
// ---------------------------------------------------------
void setup() {
  Serial.begin(115200);
  pinMode(RESET_BUTTON_PIN, INPUT_PULLUP);
  pinMode(LED_PIN, OUTPUT);
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH);  // Garante que arranca DESLIGADO (Assumindo Activo-LOW)

  WiFi.mode(WIFI_STA);
  wm.setAPCallback(configModeCallback);
  wm.setConfigPortalBlocking(false);

  Serial.println("A tentar ligação WiFi ou iniciar portal AP em background...");
  wm.autoConnect("SmartClima_Central");

  // Tenta ligar. Se falhar, abre o portal AP, mas NÃO pára o código!
  if (wm.autoConnect("SmartClima_Central")) {
    Serial.println("Ligado ao WiFi com sucesso!");
  } else {
    Serial.println("Portal AP a correr. Continua a executar o resto do código...");
  }

  // Inicia mDNS com nome estático
  if (!MDNS.begin("smartclima-central")) {
    Serial.println("Erro ao configurar mDNS");
  }

  client.setBufferSize(1024);
  client.setCallback(callback);
  lastConnectedTime = millis();
}

// ---------------------------------------------------------
// LOOP
// ---------------------------------------------------------
void loop() {
  wm.process();
  unsigned long now = millis();
  bool isResetting = (digitalRead(RESET_BUTTON_PIN) == LOW);

  // 1. LÓGICA DE RESET (Prioridade)
  if (isResetting) {
    if (tempoBotaoPressionado == 0) tempoBotaoPressionado = now;
    unsigned long decorrido = now - tempoBotaoPressionado;

    int intervalo = map(decorrido, 0, tempoParaReset, 500, 50);
    if (now - ultimoPisca > intervalo) {
      ultimoPisca = now;
      estadoLed = !estadoLed;
      digitalWrite(LED_PIN, estadoLed);
    }

    if (decorrido > tempoParaReset) {
      digitalWrite(LED_PIN, HIGH);
      wm.resetSettings();
      delay(2000);
      ESP.restart();
    }
  } else {
    tempoBotaoPressionado = 0;

    // 2. LÓGICA DO LED NORMAL
    if (wm.getConfigPortalActive() || WiFi.status() != WL_CONNECTED) {
      if (now - ultimoPisca > 500) {
        ultimoPisca = now;
        estadoLed = !estadoLed;
        digitalWrite(LED_PIN, estadoLed);
      }
    } else if (WiFi.status() == WL_CONNECTED) {
      digitalWrite(LED_PIN, client.connected() ? HIGH : LOW);
    }
  }

  // 3. CONECTIVIDADE E FAILSAFE
  if (now - lastConnectedTime > FAILSAFE_TIMEOUT) {
    if (digitalRead(RELAY_PIN) == LOW) {  // Se o relé estiver ligado (LOW)
      digitalWrite(RELAY_PIN, HIGH);      // DESLIGA (HIGH)
      client.publish(topicState, "OFF");
      Serial.println("ALERTA: Sensor ausente ou sem mensagens. Relé desligado por precaução!");
    }
  }

  if (WiFi.status() == WL_CONNECTED) {
    if (!client.connected()) {
      if (now - lastConnectedTime > 5000) {
        lastReconnectAttempt = now;
        reconnect();
      }
    } else {
      client.loop();
    }
  }
}
