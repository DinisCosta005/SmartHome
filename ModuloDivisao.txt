#include <WiFi.h>
#include <WiFiManager.h>
#include <ESPmDNS.h>
#include <PubSubClient.h>
#include <DHT.h>
#include <Preferences.h>

#define DHTPIN 4
#define DHTTYPE DHT11
#define LED_PIN 23          // LED interno para feedback
#define RESET_BUTTON_PIN 5  // AJUSTA ESTE PINO para o teu botão físico
#define DHT_INTERVAL 30000  // 30 segundos

DHT dht(DHTPIN, DHTTYPE);
Preferences preferences;
WiFiClient espClient;
PubSubClient client(espClient);
WiFiManager wm;

String roomName = "Divisao";
float setpoint = 20.0;
String currentMode = "heat";  // NOVO: Guarda o estado (heat / off)
String topicState, topicSetpointCmd, topicModeCmd;

const char* mqtt_user = "esp_mqtt";
const char* mqtt_pass = "projeto123";

unsigned long lastReadTime = 0;
unsigned long lastReconnectAttempt = 0;
unsigned long tempoBotaoPressionado = 0;
unsigned long ultimoPisca = 0;
bool estadoLed = false;
const int tempoParaReset = 5000;

// Parâmetro customizado global para ser acessível no callback
WiFiManagerParameter custom_room_name("room", "Nome da Divisao (ex: Sala)", "", 40);

// ---------------------------------------------------------
// CALLBACK: GUARDA PARÂMETROS E REINICIA
// ---------------------------------------------------------
void saveParamsCallback() {
  String novoNome = String(custom_room_name.getValue());
  if (novoNome != "" && novoNome != roomName) {
    preferences.begin("smartclima", false);
    preferences.putString("roomName", novoNome);
    preferences.end();
    Serial.println("Novo nome guardado: " + novoNome + ". A reiniciar...");
    delay(1000);
    ESP.restart();  // Reinicia para atualizar os tópicos MQTT com o novo nome
  }
}

// ---------------------------------------------------------
// CALLBACK: MODO DE CONFIGURAÇÃO (LED pisca lento)
// ---------------------------------------------------------
void configModeCallback(WiFiManager* myWiFiManager) {
  Serial.println("PORTAL DE CONFIGURAÇÃO ATIVO");
}

// ---------------------------------------------------------
// CALLBACK: RECEBE COMANDOS DO HA
// ---------------------------------------------------------
void callback(char* topic, byte* payload, unsigned int length) {
  String message;
  for (int i = 0; i < length; i++) message += (char)payload[i];

  // Recebe o Setpoint
  if (String(topic) == topicSetpointCmd) {
    setpoint = message.toFloat();
    preferences.begin("smartclima", false);
    preferences.putFloat("setpoint", setpoint);
    preferences.end();
    publishData();
  }

  // Recebe o Modo (Botão Ligar/Desligar do Home Assistant)
  if (String(topic) == topicModeCmd) {
    currentMode = message;  // "heat" ou "off"
    preferences.begin("smartclima", false);
    preferences.putString("mode", currentMode);
    preferences.end();
    publishData();
  }
}

void publishData() {
  float temp = dht.readTemperature();
  if (isnan(temp)) {
    Serial.println("ERRO CRÍTICO: Sensor DHT11 não responde!");
    // Enviamos ordem de desligar por segurança (Failsafe)
    client.publish("casa/central/rele/set", "OFF");
    
    // Avisamos o HA que a temperatura é desconhecida
    String payload = "{\"temp\": null, \"setpoint\": " + String(setpoint) + ", \"mode\": \"error\"}";
    client.publish(topicState.c_str(), payload.c_str());
    return;
  }

  // Envia os dados para o Home Assistant atualizar o interface
  String payload = "{\"temp\": " + String(temp) + ", \"setpoint\": " + String(setpoint) + ", \"mode\": \"" + currentMode + "\"}";
  client.publish(topicState.c_str(), payload.c_str());
  Serial.println("Dados publicados: " + payload);

  // --- LÓGICA DO TERMOSTATO (O CÉREBRO) ---
  if (currentMode == "off") {
    // Se o termostato estiver desligado, manda desligar o relé central
    client.publish("casa/central/rele/set", "OFF");
  } 
  else if (currentMode == "heat") {
    // Se a temperatura estiver 0.5ºC abaixo do setpoint, LIGA (histerese para não o relé não ficar a piscar)
    if (temp < (setpoint - 0.5)) {
      client.publish("casa/central/rele/set", "ON");
    } 
    // Se atingir ou ultrapassar o setpoint, DESLIGA
    else if (temp >= setpoint) {
      client.publish("casa/central/rele/set", "OFF");
    }
  }
}

void sendThermostatDiscovery() {
  String safeName = roomName;
  safeName.toLowerCase();
  safeName.replace(" ", "_");
  String discoveryTopic = "homeassistant/climate/" + safeName + "_termostato/config";

  // O JSON atualizado para ler o estado do relé central e gerar a animação!
  String payload = "{"
                   "\"name\": \"Termostato "
                   + roomName + "\","
                                "\"modes\": [\"off\", \"heat\"],"
                                "\"mode_cmd_t\": \""
                   + topicModeCmd + "\","
                                    "\"mode_stat_t\": \""
                   + topicState + "\","
                                  "\"mode_stat_tpl\": \"{{ value_json.mode }}\","
                                  "\"action_topic\": \"casa/central/rele/state\","                                    // Ouve o Relé
                                  "\"action_template\": \"{% if value == 'ON' %}heating{% else %}idle{% endif %}\","  // Cria a animação e texto
                                  "\"curr_temp_t\": \""
                   + topicState + "\","
                                  "\"curr_temp_tpl\": \"{{ value_json.temp }}\","
                                  "\"temp_stat_t\": \""
                   + topicState + "\","
                                  "\"temp_stat_tpl\": \"{{ value_json.setpoint }}\","
                                  "\"temp_cmd_t\": \""
                   + topicSetpointCmd + "\","
                                        "\"min_temp\": 15,"
                                        "\"max_temp\": 30,"
                                        "\"temp_step\": 0.5,"
                                        "\"unique_id\": \"smartclima_"
                   + safeName + "\","
                                "\"device\": {"
                                "\"ids\": [\"smartclima_sns_"
                   + safeName + "\"],"
                                "\"name\": \"Módulo "
                   + roomName + "\","
                                "\"mf\": \"SmartHome PT\""
                                "}"
                                "}";

  client.publish(discoveryTopic.c_str(), (const uint8_t*)payload.c_str(), payload.length(), true);
}

boolean reconnect() {
  if (!client.connected()) {
    Serial.println("A procurar o Home Assistant...");
    IPAddress serverIP = MDNS.queryHost("homeassistant");
    client.setServer((serverIP.toString() != "0.0.0.0") ? serverIP : IPAddress(192, 168, 1, 76), 1883);

    if (client.connect(("ESP32_Sensor_" + roomName).c_str(), mqtt_user, mqtt_pass)) {
      Serial.println("Ligado ao MQTT!");
      sendThermostatDiscovery();
      client.subscribe(topicSetpointCmd.c_str());
      client.subscribe(topicModeCmd.c_str());  // Subscreve as ordens de Ligar/Desligar
      publishData();
      return true;
    }
    Serial.println("Falhou.");
  }
  return client.connected();
}

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(LED_PIN, OUTPUT);
  pinMode(RESET_BUTTON_PIN, INPUT_PULLUP);

  preferences.begin("smartclima", false);
  roomName = preferences.getString("roomName", "Divisao");
  setpoint = preferences.getFloat("setpoint", 20.0);
  currentMode = preferences.getString("mode", "heat");  // Carrega se ficou Desligado da última vez
  preferences.end();

  WiFi.mode(WIFI_STA);  // Resolve o tal problema da memória do WiFiManager

  custom_room_name.setValue(roomName.c_str(), 40);
  wm.addParameter(&custom_room_name);
  wm.setSaveParamsCallback(saveParamsCallback);
  wm.setConfigPortalBlocking(false);

  Serial.println("A ligar ao WiFi ou iniciar portal AP...");
  if (wm.autoConnect("SmartClima_Central")) {
    Serial.println("Ligado ao WiFi!");
  }

  // Gera os tópicos baseados no roomName
  String safeTopic = roomName;
  safeTopic.toLowerCase();
  safeTopic.replace(" ", "_");

  topicState = "casa/" + safeTopic + "/state";
  topicSetpointCmd = "casa/" + safeTopic + "/setpoint/set";
  topicModeCmd = "casa/" + safeTopic + "/mode/set";

  if (!MDNS.begin(("sensor-" + safeTopic).c_str())) {
    Serial.println("Erro ao configurar mDNS");
  }

  client.setBufferSize(1024);
  client.setCallback(callback);
}

void loop() {

  wm.process();  // Essencial para o portal e parâmetros customizados
  unsigned long now = millis();

  bool isResetting = (digitalRead(RESET_BUTTON_PIN) == LOW);

  // 1. LÓGICA DO BOTÃO DE RESET (Prioridade Máxima)
  if (isResetting) {
    if (tempoBotaoPressionado == 0) tempoBotaoPressionado = now;
    unsigned long decorrido = now - tempoBotaoPressionado;

    // Pisca progressivo (de 500ms até 50ms)
    int intervalo = map(decorrido, 0, tempoParaReset, 500, 50);
    if (now - ultimoPisca > intervalo) {
      ultimoPisca = now;
      estadoLed = !estadoLed;
      digitalWrite(LED_PIN, estadoLed);
    }

    // Passaram 5 segundos -> Factory Reset
    if (decorrido > tempoParaReset) {
      Serial.println("A FAZER RESET DE FÁBRICA...");
      digitalWrite(LED_PIN, HIGH);
      wm.resetSettings();
      preferences.begin("smartclima", false);
      preferences.clear();
      preferences.end();
      delay(1500);
      ESP.restart();
    }
  } else {
    tempoBotaoPressionado = 0;  // Reset ao contador se o botão for largado

    // 2. LÓGICA DO LED NORMAL
    if (wm.getConfigPortalActive() || WiFi.status() != WL_CONNECTED) {
      // Modo de Configuração AP ou sem WiFi -> Pisca Lento
      if (now - ultimoPisca > 500) {
        ultimoPisca = now;
        estadoLed = !estadoLed;
        digitalWrite(LED_PIN, estadoLed);
      }
    } else if (WiFi.status() == WL_CONNECTED) {
      // Fixo se tiver MQTT, Apagado se o WiFi der mas o MQTT falhar
      digitalWrite(LED_PIN, client.connected() ? HIGH : LOW);
    }
  }

  // 3. GESTÃO DO MQTT E SENSORES (Apenas se não estivermos a dar reset)
  if (!isResetting && WiFi.status() == WL_CONNECTED) {
    if (!client.connected()) {
      if (now - lastReconnectAttempt > 5000) {
        lastReconnectAttempt = now;
        reconnect();
      }
    } else {
      client.loop();  // Ouve os comandos do HA

      // Lê e publica a temperatura a cada X segundos
      if (now - lastReadTime > DHT_INTERVAL) {
        lastReadTime = now;
        publishData();
      }
    }
  }
}
